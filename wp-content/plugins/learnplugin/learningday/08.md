# create database table while installing plugin

### create database table while installing plugin

```php
<?php
function learn_plugin_install()
{
    global $wpdb;
    $table_name = $wpdb->prefix . 'learn_plugin_table';
    $sql = "CREATE TABLE $table_name (
        id int(11) NOT NULL AUTO_INCREMENT,
        name varchar(255) NOT NULL,
        PRIMARY KEY  (id)
    )";
    require_once( ABSPATH . 'wp-admin/includes/upgrade.php' );
    dbDelta( $sql );
}

register_activation_hook( __FILE__, 'learn_plugin_install' );
```

### table generating code with check if table exists

function learn_plugin_create_table(){
global $wpdb;
require_once( ABSPATH . 'wp-admin/includes/upgrade.php' );

    if ( $wpdb->get_var( "SHOW TABLES LIKE '{$wpdb->prefix}learn_plugin_table'" ) != $wpdb->prefix . 'learn_plugin_table' ) {
        $sql = "CREATE TABLE {$wpdb->prefix}learn_plugin_table (
            id int(11) NOT NULL AUTO_INCREMENT,
            name varchar(255) NOT NULL,
            PRIMARY KEY  (id)
        )";
        dbDelta( $sql );
    }

}
